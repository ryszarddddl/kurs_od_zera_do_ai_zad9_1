# Używamy oficjalnego obrazu Python 3.13 (wersja slim jest lżejsza)
FROM python:3.13-slim

# Ustawiamy katalog roboczy wewnątrz kontenera
WORKDIR /app

# Instalujemy zależności systemowe (jeśli potrzebne, np. dla psycopg2)
# RUN apt-get update && apt-get install -y gcc libpq-dev && rm -rf /var/lib/apt/lists/*

# Kopiujemy plik z zależnościami
COPY requirements.txt .

# Instalujemy biblioteki Pythona
RUN pip install --no-cache-dir -r requirements.txt

# Kopiujemy resztę plików aplikacji
COPY . .

# Ustawiamy port (DigitalOcean domyślnie oczekuje 8080 lub z zmiennej env)
EXPOSE 8080

# Komenda startowa (dostosuj 'app:app' do swojej nazwy pliku i obiektu aplikacji)
# Jeśli używasz Gunicorna (zalecane na produkcji):
CMD ["gunicorn", "--bind", "0.0.0.0:8080", "app:app"]
